TROUBLESHOOTING QUICK REFERENCE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    ❌ "Can't reach database server"
       → PostgreSQL not running
       → Solution: Run START_POSTGRES_NOW.bat

    ❌ "Port 5000 already in use"
       → Another process using port 5000
       → Solution: taskkill /F /IM node.exe

    ❌ "Port 3000 already in use"
       → Another process using port 3000
       → Solution: taskkill /F /IM node.exe

    ❌ "Module not found"
       → Dependencies not installed
       → Solution: npm install (in server and client folders)

    ❌ "Chatbot not responding"
       → Check OpenAI API key in server/.env
       → Solution: Verify API key is valid


IMPORTANT REMINDERS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    ⚠️  PostgreSQL MUST be running before starting the server
    ⚠️  Keep both terminals (server and client) open while developing
    ⚠️  Don't close the terminals - they keep the services running
    ⚠️  If you close a terminal, restart that service
    ⚠️  Browser will auto-refresh when you make code changes


QUICK COMMANDS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

    Start PostgreSQL:
    net start postgresql-x64-16

    Stop PostgreSQL:
    net stop postgresql-x64-16

    Start Server:
    cd server && npm run dev

    Start Client:
    cd client && npm run dev

    Test Database:
    cd server && node -e "const {PrismaClient} = require('@prisma/client'); const p = new PrismaClient(); p.$connect().then(() => console.log('✅ Connected')).catch(e => console.log('❌', e.message))"
